        let defaultConfigFromProps = {

          // Â∞Ü props.events ‰Ωú‰∏∫ÈªòËÆ§‰∫ã‰ª∂ÈÖçÁΩÆÔºåÊ∑±Êã∑Ë¥ù‰ª•ÈÅøÂÖçÊÑèÂ§ñ‰øÆÊîπÂéüÂßã prop
          events: this.events ? JSON.parse(JSON.stringify(this.events)) : {},

        const highPriorityConfig = this.config ? JSON.parse(JSON.stringify(this.config)) : {};

        const finalConfig = mergeDeep({}, defaultConfigFromProps, highPriorityConfig);

        // 4. ÂçïÁã¨ÊûÑÂª∫‰∫ã‰ª∂Â§ÑÁêÜÂô®ÔºåÁ°Æ‰øùÊ≠£Á°ÆÁöÑ‰ºòÂÖàÁ∫ßÂíå $emit Ë°å‰∏∫
        finalConfig.events = this.buildEventHandlers();

        if (!finalConfig.document?.url || !finalConfig.document?.key) {

        if (!finalConfig.editorConfig?.callbackUrl) {

        this.$emit('ready', this.editorInstance);

     * Ê≠§ÊñπÊ≥ïÁ°Æ‰øù `config.events` ‰∏≠ÁöÑ‰∫ã‰ª∂‰ºòÂÖà‰∫é `props.events` ‰∏≠ÁöÑ‰∫ã‰ª∂Ôºå

    buildEventHandlers() {

      const self = this; // ‰øùÂ≠ò Vue ÁªÑ‰ª∂ÂÆû‰æãÁöÑÂºïÁî®
      // ÈúÄË¶Å‰ª£ÁêÜÁöÑ OnlyOffice ‰∫ã‰ª∂ÂàóË°®
      const eventsToProxy = [

        const vueEventName = eventName.substring(2).replace(/([A-Z])/g, '-$1').toLowerCase();

        // ‰∏∫ÊØè‰∏™ OnlyOffice ‰∫ã‰ª∂ÂàõÂª∫‰∏Ä‰∏™ÂåÖË£ÖÂáΩÊï∞
        handlers[eventName] = function (...eventArgs) { // ‰ΩøÁî®ÊôÆÈÄöÂáΩÊï∞‰ª•‰øùÁïô OnlyOffice ÂèØËÉΩÁöÑ `this` ‰∏ä‰∏ãÊñá
          let userHandlerReturnValue; // Â≠òÂÇ®Áî®Êà∑‰∫ã‰ª∂Â§ÑÁêÜÂô®ÁöÑËøîÂõûÂÄº
          let userHandlerCalled = false; // Ê†áËÆ∞Áî®Êà∑‰∫ã‰ª∂Â§ÑÁêÜÂô®ÊòØÂê¶Ë¢´Ë∞ÉÁî®

          // ‰ºòÂÖàÁ∫ß 1: Ê£ÄÊü• props.config.events ‰∏≠ÊòØÂê¶ÊúâÊ≠§‰∫ã‰ª∂ÁöÑÂ§ÑÁêÜÂô®
